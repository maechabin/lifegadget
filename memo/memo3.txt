
＝＝＝pm2でbabel-nodeを起動させる方法

pm2_config.json

{
  "apps" : [{
    "name"        : "lifegadget",
    "script"      : "./src/server.js",
    "watch"       : "true",
    "exec_interpreter" : "./node_modules/.bin/babel-node",
    "exec_mode"        : "fork"
  }]
}


$ pm2 start pm2_config.json
$ pm2 restart pm2_cofig.json
